<script>
function formatNumber(num) {
return num.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
}
function validateInput(input) {
if (input.value < 0) input.value = 0;
}
function calculateDues() {
const salary = parseFloat(basicSalary.value);
const years = parseFloat(serviceYears.value);
const days = parseFloat(leaveDays.value);
const alertBox = document.getElementById('alertMessage');
if (!salary || !years || !days) {
alertBox.classList.remove('hidden');
return;
}
alertBox.classList.add('hidden');
const transitionBonus = (salary * 0.16) * years;
const leaveCompensation = (salary / 30) * days;
const total = transitionBonus + leaveCompensation;
transitionBonusSpan.innerText = formatNumber(transitionBonus) + ' ريال';
leaveCompensationSpan.innerText = formatNumber(leaveCompensation) + ' ريال';
totalSum.innerText = formatNumber(total) + ' ريال';
resultsArea.classList.remove('hidden');
resultsArea.scrollIntoView({ behavior: 'smooth' });
}
function resetCalculator() {
basicSalary.value = '';
serviceYears.value = '';
leaveDays.value = '';
resultsArea.classList.add('hidden');
alertMessage.classList.add('hidden');
}
</script>
